import datetime as dt
def convert(x):
  return dt.datetime.strptime(x, '%H:%M')

def getTime(t):
    t = str(t)
    if len(t)==1:
      return t[0]+':'+'00'
    if len(t)<4:
      return t[:1] + ':' + t[1:]
    else:
      return t[:2] + ':' + t[2:]
